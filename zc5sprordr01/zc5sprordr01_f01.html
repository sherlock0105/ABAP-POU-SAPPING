FORM check_company_info .

  DATA : lv_code TYPE zc5sptcomp01-company_code,
         lv_name TYPE zc5sptcomp01-company_name.

  IF pa_code IS INITIAL.
    MESSAGE s000 WITH 'Input company code' DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

  IF pa_name IS INITIAL.
    MESSAGE s000 WITH 'Input company name' DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

  SELECT SINGLE company_code
    INTO lv_code
    FROM zc5sptcomp01
   WHERE company_code EQ pa_code.

  IF sy-subrc NE 0.
    MESSAGE s000 WITH 'Invalid data' DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

  SELECT SINGLE company_name
    INTO lv_name
    FROM zc5sptcomp01
   WHERE company_code EQ pa_code
     AND company_name EQ pa_name.

  IF sy-subrc NE 0.
    MESSAGE s000 WITH 'Invalid data' DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

ENDFORM.

FORM get_order_list .

  _clear gs_order gt_order.

  SELECT a~company_code a~company_name
         b~order_num
         b~item_id  b~order_date b~curr_qty b~meins
         b~total_amt "합계
         b~flag
         b~waers AS price_waers
         c~item_price c~waers c~item_name
    INTO CORRESPONDING FIELDS OF TABLE gt_order
    FROM zc5sptcomp01 AS a
   INNER JOIN zc5sptordr02 AS b
      ON a~company_code EQ b~company_code
    LEFT OUTER JOIN zc5sptitem01 AS c
      ON b~company_code EQ c~company_code
     AND b~item_id     EQ c~item_id
   WHERE a~company_code EQ pa_code
     AND a~company_name EQ pa_name
     AND b~flag EQ ' '
   ORDER BY b~order_num.

  gs_order-company_name = pa_name.
  gs_order-company_code = pa_code.
  gs_order-meins        = 'EA'.
  gs_order-waers        = 'KRW'.
  gs_order-order_date   = sy-datum.

  PERFORM set_order_num.

ENDFORM.

FORM save_data .

  DATA : ls_save    TYPE zc5sptordr02,
         lv_item_id TYPE zc5sptitem01-item_id.

  SELECT SINGLE item_id
    INTO lv_item_id
    FROM zc5sptitem01
   WHERE item_id EQ gs_order-item_id.

  IF sy-subrc NE 0.
    MESSAGE s000 WITH 'Invalid data' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  IF gs_order-company_code IS INITIAL OR
     gs_order-order_num IS INITIAL. "KEY 값이 없으면 EXIT.

    MESSAGE s000 WITH 'Input key field'
    DISPLAY LIKE 'E'.

    EXIT.

  ENDIF.

  CASE ' '.
    WHEN gs_order-order_num.
      MESSAGE e000 WITH 'ORDER_NUM 을 입력하세요.'.
      EXIT.
    WHEN gs_order-item_id.
      MESSAGE e000 WITH 'ITEM_ID 을 입력하세요.'.
      EXIT.
    WHEN gs_order-curr_qty.
      MESSAGE s000 WITH '수량을 입력하세요.'.
      EXIT.
  ENDCASE.

  MOVE-CORRESPONDING gs_order TO ls_save.

  IF ls_save-erdat IS INITIAL.

    ls_save-erdat = sy-datum.
    ls_save-erzet = sy-uzeit.
    ls_save-ernam = sy-uname.

  ELSE.

    ls_save-aedat = sy-datum.
    ls_save-aezet = sy-uzeit.
    ls_save-aenam = sy-uname.

  ENDIF.

  MODIFY zc5sptordr02 FROM ls_save.

  IF sy-dbcnt > 0.

    COMMIT WORK AND WAIT.
    PERFORM get_order_list.

    MESSAGE s002.

  ELSE.

    ROLLBACK WORK.
    MESSAGE s004 DISPLAY LIKE 'E'.

  ENDIF.

ENDFORM.

FORM set_color .

  DATA: lv_tabix TYPE sy-tabix,
        ls_color TYPE lvc_s_scol.

  LOOP AT gt_order INTO gs_order.

    lv_tabix = sy-tabix.

    CASE gs_order-flag.

      WHEN ' '.

        ls_color-fname     = 'FLAG'.
        ls_color-color-col = 6.
        ls_color-color-int = 1.
        ls_color-color-inv = 0.

        APPEND ls_color TO gs_order-color.
        CLEAR  ls_color.

      WHEN 'A'.

        ls_color-fname     = 'FLAG'.
        ls_color-color-col = 3.
        ls_color-color-int = 0.
        ls_color-color-inv = 0.

        APPEND ls_color TO gs_order-color.
        CLEAR  ls_color.

    ENDCASE.

    MODIFY gt_order FROM gs_order INDEX lv_tabix
    TRANSPORTING color.

  ENDLOOP.

  CLEAR gs_order.

ENDFORM.
