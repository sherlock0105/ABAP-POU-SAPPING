MODULE exit_0100 INPUT.

  IF gcl_container_1 IS BOUND.

    CALL METHOD : gcl_grid_1->free, gcl_container_1->free.
    FREE : gcl_grid_1, gcl_container_1.

  ENDIF.

  IF gcl_container_2 IS BOUND.

    CALL METHOD : gcl_grid_2->free, gcl_container_2->free.
    FREE : gcl_grid_2, gcl_container_2.

  ENDIF.

  LEAVE TO SCREEN 0.

ENDMODULE.

MODULE get_pressed INPUT.

  ENDCASE.

ENDMODULE.

MODULE user_command_0100 INPUT.

  CASE gv_okcode.

    WHEN 'REFRESH'.
      CLEAR gv_okcode.

      PERFORM get_stock.
      PERFORM set_style.
      PERFORM get_hist.
      PERFORM set_icon.
      PERFORM refresh_grid.

    WHEN 'SAVE'.
      CLEAR gv_okcode.

      PERFORM save_data.
      PERFORM get_stock.
      PERFORM set_style.
      PERFORM get_hist.
      PERFORM refresh_grid.

  ENDCASE.
ENDMODULE.
