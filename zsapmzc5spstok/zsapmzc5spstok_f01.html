FORM get_stock .

  _clear gs_stock gt_stock.

  SELECT  a~item_id       a~item_name
          a~item_cost     a~item_price  a~waers
          b~curr_qty      b~meins
          b~erdat         b~erzet       b~ernam
          b~aedat         b~aezet       b~aenam
          a~waers AS price_waers
    INTO CORRESPONDING FIELDS OF TABLE gt_stock
    FROM      zc5sptitem01 AS a
   INNER JOIN zc5sptstok01 AS b
      ON a~item_id      EQ b~item_id
   ORDER BY a~item_id.


  gt_stock_old = gt_stock.

ENDFORM.

FORM set_style .

  DATA : lv_tabix TYPE sy-tabix,
         ls_style TYPE lvc_s_styl.

  CLEAR : lv_tabix, ls_style, gs_stock.

  LOOP AT gt_stock INTO gs_stock.

    lv_tabix = sy-tabix.

    _clear ls_style gs_stock-style.

    ls_style-fieldname = 'ITEM_COST'.
    ls_style-style     = cl_gui_alv_grid=>mc_style_disabled.

    APPEND ls_style TO gs_stock-style.
    CLEAR  ls_style.

    ls_style-fieldname = 'ITEM_ID'.
    ls_style-style     = cl_gui_alv_grid=>mc_style_disabled.

    APPEND ls_style TO gs_stock-style.
    CLEAR  ls_style.

    ls_style-fieldname = 'ITEM_NAME'.
    ls_style-style     = cl_gui_alv_grid=>mc_style_disabled.

    APPEND ls_style TO gs_stock-style.
    CLEAR  ls_style.

    ls_style-fieldname = 'ITEM_PRICE'.
    ls_style-style     = cl_gui_alv_grid=>mc_style_disabled.

    APPEND ls_style TO gs_stock-style.
    CLEAR  ls_style.

    ls_style-fieldname = 'PRICE_WAERS'.
    ls_style-style     = cl_gui_alv_grid=>mc_style_disabled.

    APPEND ls_style TO gs_stock-style.
    CLEAR  ls_style.

    ls_style-fieldname = 'WAERS'.
    ls_style-style     = cl_gui_alv_grid=>mc_style_disabled.

    APPEND ls_style TO gs_stock-style.
    CLEAR  ls_style.

    MODIFY gt_stock FROM gs_stock INDEX lv_tabix
    TRANSPORTING style.

  ENDLOOP.

ENDFORM.

FORM save_data .

  DATA : ls_save_stock TYPE zc5sptstok01,
         lt_save_stock LIKE TABLE OF ls_save_stock.

  _clear ls_save_stock lt_save_stock.

  CALL METHOD gcl_grid_1->check_changed_data.


  LOOP AT gt_stock INTO gs_stock WHERE mark EQ 'X'.

    MOVE-CORRESPONDING gs_stock TO ls_save_stock.

    IF ls_save_stock-erdat IS INITIAL.
      ls_save_stock-erdat = sy-datum.
      ls_save_stock-erzet = sy-uzeit.
      ls_save_stock-ernam = sy-uname.
    ELSE.
      ls_save_stock-aedat = sy-datum.
      ls_save_stock-aezet = sy-uzeit.
      ls_save_stock-aenam = sy-uname.
    ENDIF.

    APPEND : ls_save_stock TO lt_save_stock.
    CLEAR  : ls_save_stock.

  ENDLOOP.

  IF lt_save_stock IS NOT INITIAL.

    MODIFY zc5sptstok01 FROM TABLE lt_save_stock.

  ENDIF.

  IF sy-dbcnt > 0.

    PERFORM save_hist.
    COMMIT WORK AND WAIT.
    MESSAGE s002.

  ENDIF.


ENDFORM.

FORM handle_data_changed  USING  pcl_changed TYPE REF TO cl_alv_changed_data_protocol.

  DATA : lt_modi TYPE lvc_t_modi,
         ls_modi TYPE lvc_s_modi.

  MOVE-CORRESPONDING pcl_changed->mt_mod_cells TO lt_modi.

  LOOP AT lt_modi INTO ls_modi.

    READ TABLE gt_stock INTO gs_stock INDEX ls_modi-row_id.

    IF sy-subrc NE 0.
      CONTINUE.
    ENDIF.

    gs_stock-mark = 'X'.

    MODIFY gt_stock FROM gs_stock INDEX ls_modi-row_id
    TRANSPORTING mark.

  ENDLOOP.

ENDFORM.
*&---------------------------------------------------------------------*
*& Form save_hist
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM save_hist .

  DATA : lt_hist          TYPE TABLE OF zc5sptsthi01,
         ls_hist          TYPE zc5sptsthi01,
         lv_history_lines TYPE zc5sptsthi01-stk_seq,                "hist번호
         ls_old           LIKE gs_stock,                            "과거값
         lv_diff          TYPE i.

  SELECT SINGLE MAX( stk_seq )
    INTO lv_history_lines
    FROM zc5sptsthi01.

  IF lv_history_lines IS INITIAL.
    lv_history_lines = 0.
  ENDIF.

  LOOP AT gt_stock INTO gs_stock WHERE mark = 'X'.

    CLEAR ls_old.
    READ TABLE gt_stock_old INTO ls_old
    WITH KEY item_id = gs_stock-item_id.

    IF sy-subrc <> 0.
      CONTINUE.
    ENDIF.

    lv_diff = gs_stock-curr_qty - ls_old-curr_qty.  "차이

    IF lv_diff = 0.
      CONTINUE.
    ENDIF.

    lv_history_lines = lv_history_lines + 1.

    ls_hist-stk_seq = lv_history_lines.
    ls_hist-item_id = gs_stock-item_id.
    ls_hist-chg_qty = lv_diff.
    ls_hist-meins   = gs_stock-meins.

    IF lv_diff > 0.
      ls_hist-move_type = 'IN'.
    ELSE.
      ls_hist-move_type = 'OUT'.
    ENDIF.

    ls_hist-erdat = sy-datum.
    ls_hist-erzet = sy-uzeit.
    ls_hist-ernam = sy-uname.

    APPEND ls_hist TO lt_hist.
    CLEAR ls_hist.

  ENDLOOP.

  IF lt_hist IS NOT INITIAL.
    INSERT zc5sptsthi01 FROM TABLE lt_hist.
  ENDIF.

ENDFORM.

*&      --> E_ROW_ID
*&      --> E_COLUMN_ID
*&---------------------------------------------------------------------*
FORM handle_hotspot  USING    ps_row TYPE lvc_s_row
                              ps_col TYPE lvc_s_col.

  IF ps_row-rowtype IS NOT INITIAL.
    EXIT.
  ENDIF.

  CLEAR : gs_stock.

  READ TABLE gt_stock INTO gs_stock INDEX ps_row-index.

  IF sy-subrc NE 0.
    EXIT.
  ENDIF.

  PERFORM get_pop USING gs_stock-item_id.

  IF gt_pop IS INITIAL.
    MESSAGE s000 WITH 'DATA not found'.
    EXIT.
  ENDIF.

  CALL SCREEN '0103' STARTING AT 20 3.

ENDFORM.

FORM get_pop  USING  pv_item_id TYPE zc5sptmatl01_ite-item_id.

  _clear gs_pop gt_pop.

  SELECT a~item_id    a~mat_id a~mat_txt a~mat_typ
         b~item_name
         c~labst      c~meins
    INTO CORRESPONDING FIELDS OF TABLE gt_pop
    FROM zc5sptmatl01_ite AS a
   INNER JOIN zc5sptitem01 AS b
      ON a~item_id EQ b~item_id
    INNER JOIN zc5sptmatl02 AS c
    ON a~mat_id EQ c~mat_id
   WHERE a~item_id = pv_item_id.

ENDFORM.

FORM set_icon .

  DATA : lv_tabix TYPE sy-tabix.

  LOOP AT gt_hist INTO gs_hist.
    lv_tabix = sy-tabix.

    IF gs_hist-chg_qty > 0.
      gs_hist-status = icon_positive.
    ENDIF.
    IF gs_hist-chg_qty < 0.
      gs_hist-status = icon_negative.
    ENDIF.

    MODIFY gt_hist FROM gs_hist INDEX lv_tabix
    TRANSPORTING status.

  ENDLOOP.

ENDFORM.
