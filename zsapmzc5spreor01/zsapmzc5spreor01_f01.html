FORM get_requested_order .

  _clear gs_order gt_order.

*  SELECT prod_id order_num gamng meins prode_stat erdat
*    INTO CORRESPONDING FIELDS OF TABLE gt_order
*    FROM zc5sptprod01
*   WHERE prode_stat = ' ' "변경
*   ORDER BY prod_id. "정렬 이재민

  SELECT a~prod_id a~order_num a~gamng a~meins a~prode_stat
         c~item_name a~erdat
    INTO CORRESPONDING FIELDS OF TABLE gt_order
    FROM zc5sptprod01 AS a
    INNER JOIN zc5sptordr02 AS b
       ON a~order_num = b~order_num
    LEFT OUTER JOIN zc5sptitem01 AS c
      ON b~item_id EQ c~item_id
   WHERE a~prode_stat = space
   ORDER BY a~prod_id a~order_num.

ENDFORM.

FORM make_fcat  USING   pv_key
                        pv_field
                        pv_text
                        pv_ref_table
                        pv_ref_field.

  gs_fcat-key       = pv_key.
  gs_fcat-fieldname = pv_field.
  gs_fcat-coltext   = pv_text.
  gs_fcat-ref_table = pv_ref_table.
  gs_fcat-ref_field = pv_ref_field.

  CASE pv_field.

    WHEN 'GAMNG'.
      gs_fcat-qfieldname = 'MEINS'.
    WHEN 'MEINS'.
      gs_fcat-emphasize = 'C500'.

  ENDCASE.

  APPEND gs_fcat TO gt_fcat.
  CLEAR  gs_fcat.

ENDFORM.

FORM set_style .

  DATA : lv_tabix TYPE sy-tabix,
         ls_style TYPE lvc_s_styl.

  LOOP AT gt_order INTO gs_order .

    lv_tabix = sy-tabix.

    REFRESH gs_order-style.

    ls_style-fieldname = 'PROD_ID'.
    ls_style-style     = cl_gui_alv_grid=>mc_style_button.

    APPEND ls_style TO gs_order-style.
    CLEAR  ls_style.

    MODIFY gt_order FROM gs_order INDEX lv_tabix
    TRANSPORTING style.

  ENDLOOP.

ENDFORM.
