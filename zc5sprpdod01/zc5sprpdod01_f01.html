FORM get_prod_order .

  _clear gs_prod_order gt_prod_order.

  GET PARAMETER ID : 'ORN' FIELD gs_prod_order-order_num,
                     'INM' FIELD gs_prod_order-item_name.

  SELECT a~prod_id a~order_num a~gamng a~meins a~prode_stat a~erdat AS order_date
         c~item_name
    INTO CORRESPONDING FIELDS OF TABLE gt_prod_order
    FROM zc5sptprod01 AS a
    INNER JOIN zc5sptordr02 AS b
       ON a~order_num = b~order_num
    LEFT OUTER JOIN zc5sptitem01 AS c
      ON b~item_id EQ c~item_id
   WHERE a~prode_stat = space
   ORDER BY a~prod_id a~order_num.

  gs_prod_order-meins = 'EA'. "이재민

ENDFORM.

FORM save_data .

  DATA : lv_prod_id TYPE zc5sptprod01-prod_id,
         ls_save    TYPE zc5sptprod01.

  CLEAR : lv_prod_id, ls_save.

* 필수값 체크
  IF gs_prod_order-order_num IS INITIAL.

    MESSAGE s000 WITH TEXT-e01 DISPLAY LIKE 'E'.
    EXIT.

  ENDIF.

* 수정가능한지 체크
  IF gs_prod_order-prode_stat IS NOT INITIAL.

    MESSAGE s000 WITH TEXT-e03 DISPLAY LIKE 'E'.
    EXIT.

  ENDIF.

  IF gs_prod_order-prod_id IS INITIAL.

    PERFORM get_new_prod_id CHANGING lv_prod_id.
    gs_prod_order-prod_id = lv_prod_id.

  ENDIF.

  MOVE-CORRESPONDING gs_prod_order TO ls_save.

  IF ls_save-erdat IS INITIAL.

    ls_save-erdat = sy-datum.
    ls_save-erzet = sy-uzeit.
    ls_save-ernam = sy-uname.

  ELSE.

    ls_save-aedat = sy-datum.
    ls_save-aezet = sy-uzeit.
    ls_save-aenam = sy-uname.

  ENDIF.

  IF ls_save IS INITIAL.
    EXIT.
  ENDIF.

  MODIFY zc5sptprod01 FROM ls_save.

  IF sy-dbcnt > 0.
    COMMIT WORK AND WAIT.

    PERFORM get_prod_order.
    PERFORM refresh_grid.

    MESSAGE s002.
  ELSE.
    ROLLBACK WORK.
    MESSAGE s004 DISPLAY LIKE 'E'.
  ENDIF.

ENDFORM.

FORM get_new_prod_id CHANGING pv_prod_id TYPE zc5sptprod01-prod_id.

  DATA lv_max_id TYPE zc5sptprod01-prod_id.
  DATA lv_next   TYPE i.

  SELECT MAX( prod_id )
    INTO lv_max_id
    FROM zc5sptprod01.

  IF lv_max_id IS INITIAL.

    pv_prod_id = 1.

  ELSE.

    lv_next = lv_max_id.
    lv_next = lv_next + 1.
    pv_prod_id = lv_next.

  ENDIF.

ENDFORM.

FORM f4_order_num .

  DATA : BEGIN OF ls_order,
           order_num    TYPE zc5sptordr02-order_num,
           company_name TYPE zc5sptcomp01-company_name,
           item_name    TYPE zc5sptitem01-item_name,
           curr_qty     TYPE zc5sptordr02-curr_qty,
           meins        TYPE zc5sptordr02-meins,
           order_date   TYPE zc5sptordr02-order_date,
         END OF ls_order,

         lt_order  LIKE TABLE OF ls_order,

         ls_return TYPE ddshretval,
         lt_return LIKE TABLE OF ls_return.

  _clear ls_order  lt_order.
  _clear ls_return lt_return.

  SELECT a~company_name
         b~order_num     b~curr_qty b~meins b~order_date
         c~item_name
    INTO CORRESPONDING FIELDS OF TABLE lt_order
    FROM zc5sptcomp01 AS a
   INNER JOIN zc5sptordr02 AS b
      ON a~company_code EQ b~company_code
    LEFT OUTER JOIN zc5sptitem01 AS c
      ON b~item_id EQ c~item_id
   WHERE b~flag = 'A'
   ORDER BY b~order_num.

  CALL FUNCTION 'F4IF_INT_TABLE_VALUE_REQUEST'
    EXPORTING
      retfield     = 'ORDER_NUM'
      dynpprog     = sy-repid
      dynpnr       = sy-dynnr
      window_title = 'Order List'
      value_org    = 'S'
    TABLES
      return_tab   = lt_return
      value_tab    = lt_order.

  READ TABLE lt_return INTO ls_return INDEX 1.

  IF sy-subrc NE 0.
    EXIT.
  ENDIF.

  gs_prod_order-order_num = ls_return-fieldval.

  READ TABLE lt_order INTO ls_order WITH KEY order_num = gs_prod_order-order_num.

  IF sy-subrc EQ 0.

    CALL FUNCTION 'SAPGUI_SET_FUNCTIONCODE'
      EXPORTING
        functioncode           = 'SETORDERDATA'
      EXCEPTIONS
        function_not_supported = 1
        OTHERS                 = 2.

  ENDIF.

ENDFORM.

FORM handle_hotspot  USING ps_row TYPE lvc_s_row
                           ps_col TYPE lvc_s_col.

  IF ps_row-rowtype IS NOT INITIAL.
    EXIT.
  ENDIF.

  READ TABLE gt_prod_order INTO gs_prod_order INDEX ps_row-index.

  IF sy-subrc NE 0.
    EXIT.
  ENDIF.

  CASE ps_col-fieldname.
    WHEN 'ORDER_NUM'.

      PERFORM get_order_pop USING gs_prod_order-order_num.

      IF gt_order_pop IS INITIAL.
        MESSAGE s000 WITH TEXT-e04.
        EXIT.
      ENDIF.

      CALL SCREEN '0101' STARTING AT 20 3.

  ENDCASE.

ENDFORM.

FORM get_order_pop  USING pv_order_num.

  _clear gs_order_pop gt_order_pop.

  SELECT a~company_code a~company_name
         b~order_num
         b~item_id  b~order_date b~curr_qty b~meins
         b~total_amt "합계
         b~flag
         c~item_price c~waers c~item_name
    INTO CORRESPONDING FIELDS OF TABLE gt_order_pop
    FROM zc5sptcomp01 AS a
   INNER JOIN zc5sptordr02 AS b
      ON a~company_code EQ b~company_code
    LEFT OUTER JOIN zc5sptitem01 AS c
      ON b~company_code EQ c~company_code
     AND b~item_id      EQ c~item_id
   WHERE b~order_num    EQ pv_order_num.
