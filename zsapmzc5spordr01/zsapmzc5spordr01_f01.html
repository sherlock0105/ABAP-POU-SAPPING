FORM make_fcat  USING   pv_key
                        pv_field
                        pv_text
                        pv_ref_table
                        pv_ref_field.

  gs_fcat-key       = pv_key.
  gs_fcat-fieldname = pv_field.
  gs_fcat-coltext   = pv_text.
  gs_fcat-ref_table = pv_ref_table.
  gs_fcat-ref_field = pv_ref_field.

  CASE pv_field.

    WHEN 'TOTAL_AMT' OR 'ITEM_PRICE'.
      gs_fcat-cfieldname = 'WAERS'.

    WHEN 'CURR_QTY'.
      gs_fcat-qfieldname = 'MEINS'.

    WHEN 'MEINS' OR 'WAERS' OR 'PRICE_WAERS'.
      gs_fcat-emphasize  = 'C500'.

  ENDCASE.

  APPEND gs_fcat TO gt_fcat.
  CLEAR  gs_fcat.

ENDFORM.

FORM get_order_list .

  DATA : lv_order_num TYPE zc5sptordr02-order_num.

  SELECT a~company_code a~company_name
         b~order_num
         b~item_id  b~order_date b~curr_qty b~meins
         b~total_amt "합계
         b~flag
         b~waers AS price_waers
         c~item_price c~waers c~item_name
         c~item_cost
    INTO CORRESPONDING FIELDS OF TABLE gt_order
    FROM zc5sptcomp01 AS a
   INNER JOIN zc5sptordr02 AS b
      ON a~company_code EQ b~company_code
    LEFT OUTER JOIN zc5sptitem01 AS c
      ON b~company_code EQ c~company_code
     AND b~item_id     EQ c~item_id
   WHERE b~flag NOT IN ( 'P' , 'A' )
   ORDER BY a~company_code b~order_num.

ENDFORM.

  DATA : lv_tabix TYPE sy-tabix,
         ls_color TYPE lvc_s_scol.

  LOOP AT gt_order INTO gs_order.

    lv_tabix = sy-tabix.

    IF gs_order-flag = 'A'.

    ENDIF.

    CASE gs_order-flag.

      WHEN 'A'.

        ls_color-fname = 'FLAG'.
        ls_color-color-col = 5.
        ls_color-color-int = 1.
        ls_color-color-inv = 1.

        APPEND ls_color TO gs_order-color.
        CLEAR  ls_color.

      WHEN 'R'.

        ls_color-fname = 'FLAG'.
        ls_color-color-col = 6.
        ls_color-color-int = 1.
        ls_color-color-inv = 1.

        APPEND ls_color TO gs_order-color.
        CLEAR  ls_color.

      WHEN ' '.

        ls_color-fname = 'FLAG'.
        ls_color-color-col = 3.
        ls_color-color-int = 1.
        ls_color-color-inv = 1.

        APPEND ls_color TO gs_order-color.
        CLEAR  ls_color.

    ENDCASE.

    MODIFY gt_order FROM gs_order INDEX lv_tabix
    TRANSPORTING color.

  ENDLOOP.

ENDFORM.

FORM approve_order_multi .

  DATA: lv_answer,
        lv_question(50).

  CLEAR gt_row.

  CALL METHOD gcl_grid->get_selected_rows
    IMPORTING
      et_index_rows = gt_row.

  IF gt_row IS INITIAL.
    MESSAGE s000 WITH '승인할 행을 선택하세요.' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.


  LOOP AT gt_row INTO gs_row.

    READ TABLE gt_order INTO gs_order INDEX gs_row-index.
    CHECK sy-subrc = 0.

    UPDATE zc5sptordr02
       SET flag = 'A'
     WHERE order_num = gs_order-order_num.

    IF sy-subrc <> 0.
      ROLLBACK WORK.
      MESSAGE s000 WITH '승인 처리 중 오류가 발생했습니다.'
      DISPLAY LIKE 'E'.
      EXIT.
    ENDIF.

  ENDLOOP.

  COMMIT WORK.

  LOOP AT gt_row INTO gs_row.
    READ TABLE gt_order INTO gs_order INDEX gs_row-index.
    IF sy-subrc = 0.
      gs_order-flag = 'A'.
      MODIFY gt_order FROM gs_order INDEX gs_row-index.
    ENDIF.
  ENDLOOP.

  DATA lv_msg TYPE string.

  lv_msg = |{ lines( gt_row ) }건이 승인되었습니다.|.

  MESSAGE s000 WITH lv_msg.

  PERFORM get_order_list.
  PERFORM style_set.
  PERFORM refresh_grid.

ENDFORM.
