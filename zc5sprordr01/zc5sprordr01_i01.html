MODULE exit_0100 INPUT.

  CALL METHOD : gcl_grid->free, gcl_container->free.

  FREE        : gcl_grid,       gcl_container.

  CASE gv_okcode.

    WHEN 'EXIT'.
      LEAVE PROGRAM.

    WHEN OTHERS.
      LEAVE TO SCREEN 0.

  ENDCASE.

ENDMODULE.

MODULE user_command_0100 INPUT.

  CASE gv_okcode.

    WHEN 'REFRESH'.
      CLEAR gv_okcode.

      PERFORM get_order_list.
      PERFORM refresh_grid.

    WHEN 'DISPLAY'.

      PERFORM display_data.

    WHEN 'APPROVAL'.
      CLEAR gv_okcode.

      PERFORM payment.

    WHEN 'SAVE'.
      CLEAR gv_okcode.

      PERFORM save_data.
      PERFORM refresh_grid.

    WHEN 'DELETE'.
      CLEAR gv_okcode.
      PERFORM delete_row.

    WHEN 'SETNAME'.
      PERFORM set_name.

  ENDCASE.

ENDMODULE.

MODULE f4_item_id INPUT.

  DATA : BEGIN OF ls_item,
           item_id    TYPE zc5sptitem01-item_id,
           item_name  TYPE zc5sptitem01-item_name,
           item_price TYPE zc5sptitem01-item_price,
           waers      TYPE zc5sptitem01-waers,
         END OF ls_item,

         lt_item LIKE TABLE OF ls_item.

  DATA : ls_return TYPE ddshretval,
         lt_return LIKE TABLE OF ls_return.

  _clear ls_item lt_item.
  _clear ls_return lt_return.

  SELECT item_id item_name item_price waers
    INTO CORRESPONDING FIELDS OF TABLE lt_item
    FROM zc5sptitem01
   WHERE company_code EQ pa_code
   ORDER BY item_id.

  CALL FUNCTION 'F4IF_INT_TABLE_VALUE_REQUEST'
    EXPORTING
      retfield     = 'ITEM_ID'
      dynpprog     = sy-repid
      dynpnr       = sy-dynnr
      window_title = 'ITEM_NAME'
      value_org    = 'S'
    TABLES
      return_tab   = lt_return
      value_tab    = lt_item.

  READ TABLE lt_return INTO ls_return INDEX 1.

  IF sy-subrc NE 0.
    EXIT.
  ENDIF.

  gs_order-item_id = ls_return-fieldval.

  READ TABLE lt_item INTO ls_item WITH KEY item_id = gs_order-item_id.

  IF sy-subrc EQ 0.

    CALL FUNCTION 'SAPGUI_SET_FUNCTIONCODE'
      EXPORTING
        functioncode           = 'SETNAME'
      EXCEPTIONS
        function_not_supported = 1
        OTHERS                 = 2.

  ENDIF.

ENDMODULE.
