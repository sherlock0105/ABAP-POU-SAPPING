  FREE : gcl_grid, gcl_container.

  LEAVE TO SCREEN 0.

ENDMODULE.

MODULE user_command_0100 INPUT.

  CASE gv_okcode.
    WHEN 'PROD_ORD'. " 상품 생산 요청
      CLEAR gv_okcode.
      PERFORM prod_order.
*      PERFORM refresh_grid.

    WHEN 'MAT_ORD'. " 자재 생산 요청
      CLEAR gv_okcode.
      PERFORM mat_order.

    WHEN 'REFRESH'.
      PERFORM get_material.
      PERFORM set_material.
      IF gv_flag IS NOT INITIAL.
        PERFORM set_after_prod.
      ENDIF.
      PERFORM refresh_grid.

  ENDCASE.

ENDMODULE.
